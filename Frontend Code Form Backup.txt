<script>
  const form = document.getElementById("repleafForm");
  const statusMsg = document.getElementById("statusMsg");
  const step1 = document.getElementById("step1");
  const step2 = document.getElementById("step2");
  const nextBtn = document.getElementById("nextBtn");
  const emailNote = document.getElementById("email-note");

  nextBtn.addEventListener("click", () => {
    let isValid = true;
    step1.querySelectorAll(".form-group").forEach((group) => {
      const input = group.querySelector(".form-input");
      const error = group.querySelector(".error-msg");
      if (!input.value.trim()) {
        error.classList.add("active");
        isValid = false;
      } else {
        error.classList.remove("active");
      }
    });
    if (!isValid) return;

    step1.classList.remove("active");
    step2.classList.add("active");
    emailNote.style.display = "block";
  });

  form.addEventListener("submit", async function (e) {
    e.preventDefault();
    let isValid = true;
    step2.querySelectorAll(".form-group").forEach((group) => {
      const input = group.querySelector(".form-input");
      const error = group.querySelector(".error-msg");
      if (
        !input.value.trim() ||
        (input.type === "email" && !/\S+@\S+\.\S+/.test(input.value))
      ) {
        error.classList.add("active");
        isValid = false;
      } else {
        error.classList.remove("active");
      }
    });
    if (!isValid) return;

    statusMsg.innerText = "Let AI write its magic...";
    try {
      const response = await fetch(
        "https://repleaf-website-backend.vercel.app/api/send-email",
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            firstName: document.getElementById("firstName").value,
            lastName: document.getElementById("lastName").value,
            email: document.getElementById("email").value,
            headline: document.getElementById("headline").value,
          }),
        }
      );
      const result = await response.json();
      if (result.success) {
        statusMsg.innerText =
          "‚úÖ Done. Check your email for your personalized Repleaf guide!";
      } else {
        statusMsg.innerText = "‚ùå Something went wrong. Please try again.";
      }
    } catch (err) {
      console.error(err);
      statusMsg.innerText = "‚ö†Ô∏è Server error. Please try again later.";
    }
  });
</script>
